# 입력
> 첫째 줄에 A, B, C가 빈 칸을 사이에 두고 순서대로 주어진다. A, B, C는 21억 이하의 자연수이다.

-----
### 나의 처음 코드

~~~python
# a(고정비용), b(가변비용), c(노트북 가격)
# a/(c-b) < x
import math
a, b, c = list(map(int, input().split()))

if (c-b) > 0: #분모가 0보다 커야 손익분기점을 가질 수 있다.
    fomula = a/(c-b)
    print(int(fomula) + 1)  # math.ceil() = 반올림
else:
    print("-1")
~~~
